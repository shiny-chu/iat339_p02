	display: block;
	font-weight: 600;
	text-transform: uppercase;
	color: rgb(114, 114, 114);
	margin: 0 0 0.1rem 0.1rem;
}

	width: 100%;
	padding: 0.8rem 1rem;
	border: 2px solid rgb(169, 169, 169);
	transition: all 0.3s ease 0s;
}
input[type=text]:focus{
	border: 2px solid rgb(199, 168, 134);
	outline: none;
}
